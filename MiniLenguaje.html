//    declaracion_variables() declaracion_acciones() bloque_sentencias()
programa ::= <tPROGRAMA> <tIDENTIFICADOR> <tFIN_SENTENCIA> declaracion_variables declaracion_acciones bloque_sentencias <EOF> 
// (declaracion())*
declaracion_variables ::= ( declaracion <tFIN_SENTENCIA> )* 
// tipo_variables() identificadores()
declaracion ::= tipo_variables identificadores 
//  |  |  |
tipo_variables ::= <tENTERO>
| <tCARACTER>
| <tBOOLEANO>
| <tDECIMAL> 
// ()*
identificadores ::= <tIDENTIFICADOR> ( <tCORCHETE_IZQ> <tVALOR_ENTERO> <tCORCHETE_DCHA> )? ( <tCOMA> <tIDENTIFICADOR> ( <tCORCHETE_IZQ> <tVALOR_ENTERO> <tCORCHETE_DCHA> )? )* 
// (declaracion_accion())*
declaracion_acciones ::= ( declaracion_accion )* 
// cabecera_accion()  declaracion_variables() declaracion_acciones() bloque_sentencias()
declaracion_accion ::= cabecera_accion <tFIN_SENTENCIA> declaracion_variables declaracion_acciones bloque_sentencias 
//   parametros_formales()
cabecera_accion ::= <tACCION> <tIDENTIFICADOR> parametros_formales 
// (( parametros()) ( parametros())* )?
parametros_formales ::= ( ( <tPARENTESIS_IZQ> parametros ) ( <tFIN_SENTENCIA> parametros )* <tPARENTESIS_DCHA> )? 
// expresion() ( expresion())*
lista_parametros ::= identificadores 
// clase_parametros() tipo_variables() lista_parametros()
parametros ::= clase_parametros tipo_variables lista_parametros 
//  |
clase_parametros ::= <tVAL>
| <tREF> 
//  lista_sentencias()
bloque_sentencias ::= <tPRINCIPIO> lista_sentencias <tFIN> 
// (sentencia())*
lista_sentencias ::= ( sentencia )+ 
/*
leer()  < tFIN_SENTENCIA >|
  escribir() < tFIN_SENTENCIA >|
  < tIDENTIFICADOR > [asignacion() |  invocacion_accion() ] < tFIN_SENTENCIA > |
  seleccion()  |
  mientras_que()
  */
sentencia ::= ( leer <tFIN_SENTENCIA> | escribir <tFIN_SENTENCIA> | <tIDENTIFICADOR> ( asignacion | invocacion_accion )? | seleccion | mientras_que ) 
//   lista_asignables()
leer ::= <tLEER> <tPARENTESIS_IZQ> lista_asignables <tPARENTESIS_DCHA> 
//  ( )*
lista_asignables ::= <tIDENTIFICADOR> ( <tCOMA> <tIDENTIFICADOR> )* 
//   lista_escribibles()
escribir ::= <tESCRIBIR> <tPARENTESIS_IZQ> lista_escribibles <tPARENTESIS_DCHA> 
// expresion() ( expresion())*
lista_escribibles ::= ( expresion | <tSTRING> ) ( <tCOMA> ( expresion | <tSTRING> ) )* 
//  expresion()
asignacion ::= ( <tCORCHETE_IZQ> expresion <tCORCHETE_DCHA> )? <tOPAS> expresion <tFIN_SENTENCIA> 
// argumentos()
invocacion_accion ::= ( argumentos )? <tFIN_SENTENCIA> 
//  expresion() lista_sentencias()
mientras_que ::= <tMQ> expresion lista_sentencias <tFMQ> 
//  expresion()  lista_sentencias() ( lista_sentencias())?
seleccion ::= <tSI> expresion <tENT> lista_sentencias ( <tSI_NO> lista_sentencias )? <tFSI> 
//  (lista_expresiones())?
argumentos ::= <tPARENTESIS_IZQ> ( lista_expresiones )? <tPARENTESIS_DCHA> 
// expresion() ( expresion())*
lista_expresiones ::= expresion ( <tCOMA> expresion )* 
// expresion_simple() (operador_relacional() expresion_simple())*
expresion ::= expresion_simple ( operador_relacional expresion_simple )? 
//  |  |  |  |  |
operador_relacional ::= <tMAYOR>
| <tMENOR>
| <tMAI>
| <tMEI>
| <tNI>
| <tIGUAL> 
// termino() (operador_aditivo() termino())*
expresion_simple ::= termino ( operador_aditivo termino )* 
//  |  |
operador_aditivo ::= <tMAS>
| <tMENOS>
| <tOR> 
// factor() (operador_multiplicativo() factor())*
termino ::= factor ( operador_multiplicativo factor )* 
//  |  |  |
operador_multiplicativo ::= <tPOR>
| <tDIV>
| <tMOD>
| <tAND>

	< tMENOS > factor()
| < tNOT > factor()
| < tPARENTESIS_IZQ > expresion() < tPARENTESIS_DCHA >
| < tENTACAR >  expresion() < tPARENTESIS_DCHA >
